function palindrome(str) {
  var cleanStr = str.replace(/[^A-Z\dαινσϊδλοφόρ]/ig, '').toLowerCase();
  function rev(str) { return str.split('').reverse().join(''); }

  return (cleanStr === rev(cleanStr));
}
